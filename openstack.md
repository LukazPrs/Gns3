>    nano glanceapi.te

   module glanceapi 1.0;

require {
	type glance_api_t;
        type httpd_config_t;
        type iscsid_exec_t;
        class dir search;
        class file { getattr open read };
}

#============= glance_api_t =====
allow glance_api_t httpd_config_t:dir search;
allow glance_api_t iscsid_exec_t:file { getattr open read };
